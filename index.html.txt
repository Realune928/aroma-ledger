function getMonths() {
  const months = new Set();
  months.add(`${now.getFullYear()}-${String(now.getMonth()+1).padStart(2,'0')}`);
  allEntries.forEach(e => months.add(e.date.slice(0,7)));
  return [...months].sort().reverse();
}

function buildMonthSelect() {
  const sel = document.getElementById('month-select');
  const months = getMonths();
  sel.innerHTML = months.map(m => {
    const [y,mo] = m.split('-');
    return `<option value="${m}">${y}年${parseInt(mo)}月</option>`;
  }).join('');
  if (!currentMonth || !months.includes(currentMonth)) currentMonth = months[0];
  sel.value = currentMonth;
  const [y,mo] = currentMonth.split('-');
  document.getElementById('month-label').textContent = `${y}年${parseInt(mo)}月`;
}

function fmt(n) {
  return '¥' + n.toLocaleString('zh-CN', { minimumFractionDigits: 0, maximumFractionDigits: 0 });
}

function parseLine(line) {
  line = line.trim();
  if (!line) return null;
  const m = line.match(/^(.+?)\s+(\d+(?:\.\d+)?)(?:[×x\*](\d+(?:\.\d+)?))?/i);
  if (!m) return null;
  const name  = m[1].trim();
  const price = parseFloat(m[2]);
  const qty   = m[3] ? parseFloat(m[3]) : 1;
  return { name, price, qty, amount: price * qty };
}

> <script>
> const CAT = {
>  service:  { label: '服务 · 疗程', dot: 'dot-service' },
>  product:  { label: '产品 · 配方', dot: 'dot-product' },
>  course:   { label: '课程 · 工作坊', dot: 'dot-course' },
>  consult:  { label: '咨询 · 个案', dot: 'dot-consult' },
>  other:    { label: '其他', dot: 'dot-other' },
> };
> 
> let allEntries = JSON.parse(localStorage.getItem('aroma_entries') || '[]');
> let currentMonth = '';
> 
> const now = new Date();
> document.getElementById('today-date').textContent =
>  now.toLocaleDateString('zh-CN', { year:'numeric', month:'long', day:'numeric', weekday:'long' });
> document.getElementById('f-date').value = now.toISOString().slice(0,10);
> document.getElementById('b-date').value = now.toISOString().slice(0,10);
> 
> function save() {
>  localStorage.setItem('aroma_entries', JSON.stringify(allEntries));
> }
> 
> function getMonths() {
>  const months = new Set();
>  const y = now.getFullYear();
>  const m = String(now.getMonth()+1).padStart(2,'0');
>  months.add(`${y}-${m}`);
>  allEntries.forEach(e => months.add(e.date.slice(0,7)));
>  return [...months].sort().reverse();
> }
> 
> function buildMonthSelect() {
>  const sel = document.getElementById('month-select');
>  const months = getMonths();
>  sel.innerHTML = months.map(m => {
>    const [y,mo] = m.split('-');
>    return `<option value="${m}">${y}年${parseInt(mo)}月</option>`;
>  }).join('');
>  if (!currentMonth || !months.includes(currentMonth)) currentMonth = months[0];
>  sel.value = currentMonth;
>  const [y,mo] = currentMonth.split('-');
>  document.getElementById('month-label').textContent = `${y}年${parseInt(mo)}月`;
> }
> 
> function changeMonth() {
>  currentMonth = document.getElementById('month-select').value;
>  const [y,mo] = currentMonth.split('-');
>  document.getElementById('month-label').textContent = `${y}年${parseInt(mo)}月`;
>  render();
> }
> 
> function toggleForm() {
>  document.getElementById('add-form').classList.toggle('visible');
>  document.getElementById('add-toggle').classList.toggle('open');
> }
> 
> function toggleBatch() {
>  document.getElementById('batch-form').classList.toggle('visible');
>  document.getElementById('batch-toggle').classList.toggle('open');
> }
> 
> function addEntry() {
>  const date   = document.getElementById('f-date').value;
>  const amount = parseFloat(document.getElementById('f-amount').value);
>  const cat    = document.getElementById('f-cat').value;
>  const name   = document.getElementById('f-name').value.trim();
>  const note   = document.getElementById('f-note').value.trim();
>  if (!date || !amount || amount <= 0) { alert('请填写日期和金额'); return; }
>  allEntries.push({ id: Date.now(), date, amount, cat, name, note });
>  allEntries.sort((a,b) => b.date.localeCompare(a.date));
>  save();
>  document.getElementById('f-amount').value = '';
>  document.getElementById('f-name').value = '';
>  document.getElementById('f-note').value = '';
>  currentMonth = date.slice(0,7);
>  buildMonthSelect();
>  render();
>  toggleForm();
> }
> 
> function deleteEntry(id) {
>  if (!confirm('删除这笔记录？')) return;
>  allEntries = allEntries.filter(e => e.id !== id);
>  save();
>  buildMonthSelect();
>  render();
> }
> 
> function fmt(n) {
>  return '¥' + n.toLocaleString('zh-CN', { minimumFractionDigits: 0, maximumFractionDigits: 0 });
> }
> 
> function parseLine(line) {
>  line = line.trim();
>  if (!line) return null;
>  const m = line.match(/^(.+?)\s+(\d+(?:\.\d+)?)(?:[×x\*](\d+(?:\.\d+)?))?/i);
>  if (!m) return null;
>  const name  = m[1].trim();
>  const price = parseFloat(m[2]);
>  const qty   = m[3] ? parseFloat(m[3]) : 1;
>  return { name, price, qty, amount: price * qty };
> }
> 
> let batchParsed = [];
> 
> function parseBatch() {
>  const lines = document.getElementById('b-text').value.split('\n').filter(l => l.trim());
>  batchParsed = lines.map(parseLine).filter(Boolean);
>  const preview = document.getElementById('b-preview');
>  const btn     = document.getElementById('b-submit');
>  if (!batchParsed.length) { preview.classList.remove('visible'); btn.disabled = true; return; }
>  const total = batchParsed.reduce((s,i) => s + i.amount, 0);
>  preview.classList.add('visible');
>  preview.innerHTML = batchParsed.map(i => `
>    <div class="preview-item">
>      <span class="preview-name">${i.name}${i.qty > 1 ? ' ×'+i.qty : ''}</span>
>      <span class="preview-amt">¥${i.amount.toLocaleString()}</span>
>    </div>
>  `).join('') + `
>    <div class="preview-total">
>      <span>合计 ${batchParsed.length} 笔</span>
>      <span>¥${total.toLocaleString()}</span>
>    </div>`;
>  btn.disabled = false;
> }
> 
> function submitBatch() {
>  if (!batchParsed.length) return;
>  const date   = document.getElementById('b-date').value;
>  const client = document.getElementById('b-client').value.trim();
>  const cat    = document.getElementById('b-cat').value;
>  if (!date) { alert('请填写日期'); return; }
>  batchParsed.forEach((item, i) => {
>    allEntries.push({
>      id: Date.now() + i,
>      date, amount: item.amount, cat,
>      name: client ? `${client} · ${item.name}` : item.name,
>      note: item.qty > 1 ? `单价 ¥${item.price} × ${item.qty}` : '',
>    });
>  });
>  allEntries.sort((a,b) => b.date.localeCompare(a.date));
>  save();
>  document.getElementById('b-text').value = '';
>  document.getElementById('b-client').value = '';
>  document.getElementById('b-preview').classList.remove('visible');
>  document.getElementById('b-submit').disabled = true;
>  batchParsed = [];
>  currentMonth = date.slice(0,7);
>  buildMonthSelect();
>  render();
>  toggleBatch();
> }
> 
> function render() {
>  const curY = currentMonth.slice(0,4);
>  const monthItems = allEntries.filter(e => e.date.slice(0,7) === currentMonth);
>  const yearItems  = allEntries.filter(e => e.date.slice(0,4) === curY);
>  const mTotal = monthItems.reduce((s,e) => s+e.amount, 0);
>  const yTotal = yearItems.reduce((s,e) => s+e.amount, 0);
>  const avg    = monthItems.length ? mTotal / monthItems.length : 0;
>  document.getElementById('s-month').textContent = fmt(mTotal);
>  document.getElementById('s-month-count').textContent = monthItems.length + ' 笔';
>  document.getElementById('s-year').textContent = fmt(yTotal);
>  document.getElementById('s-year-count').textContent = yearItems.length + ' 笔';
>  document.getElementById('s-avg').textContent = fmt(avg);
>  const catTotals = {};
>  monthItems.forEach(e => { catTotals[e.cat] = (catTotals[e.cat] || 0) + e.amount; });
>  const catBars = document.getElementById('cat-bars');
>  if (!Object.keys(catTotals).length) {
>    catBars.innerHTML = '<div style="font-size:12px;color:var(--text-soft);text-align:center;padding:8px 0">暂无数据</div>';
>  } else {
>    catBars.innerHTML = Object.entries(catTotals).sort((a,b) => b[1]-a[1]).map(([cat, amt]) => `
>      <div class="cat-row">
>        <div class="cat-meta">
>          <span class="cat-name">${CAT[cat].label}</span>
>          <span class="cat-amt">${fmt(amt)}</span>
>        </div>
>        <div class="bar-track">
>          <div class="bar-fill" style="width:${(amt/mTotal*100).toFixed(1)}%"></div>
>        </div>
>      </div>`).join('');
>  }
>  const list = document.getElementById('records-list');
>  document.getElementById('rec-count').textContent = monthItems.length + ' 笔';
>  if (!monthItems.length) {
>    list.innerHTML = `<div class="empty-state"><div class="emoji">🌿</div>这个月还没有记录<br>记录第一笔收入吧</div>`;
>    return;
>  }
>  list.innerHTML = monthItems.map(e => {
>    const c = CAT[e.cat];
>    const d = new Date(e.date + 'T00:00:00');
>    const dateStr = d.toLocaleDateString('zh-CN', { month:'numeric', day:'numeric', weekday:'short' });
>    return `
>      <div class="record-item">
>        <div class="record-dot ${c.dot}"></div>
>        <div class="record-body">
>          <div class="record-top">
>            <div class="record-title">${e.name || c.label}</div>
>            <div class="record-amount">+${fmt(e.amount)}</div>
>          </div>
>          <div class="record-meta">
>            <span>${dateStr}</span>
>            <span>${c.label}</span>
>          </div>
>          ${e.note ? `<div class="record-note">${e.note}</div>` : ''}
>        </div>
>        <button class="btn-del" onclick="deleteEntry(${e.id})">×</button>
>      </div>`;
>  }).join('');
> }
